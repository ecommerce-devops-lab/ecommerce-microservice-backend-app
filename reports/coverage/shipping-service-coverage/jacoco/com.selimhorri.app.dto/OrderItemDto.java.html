package com.selimhorri.app.dto;

import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Data;
import lombok.NoArgsConstructor;

@AllArgsConstructor
@Data
@Builder

	private Integer orderId;
	private Integer orderedQuantity;

	private OrderDto orderDto;
	
}
